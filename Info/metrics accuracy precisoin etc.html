import matplotlib.pyplot as plt
import pydicom
from PIL import Image
import os

def display_images_with_annotations(df, images_per_row=1, rows=1, output_dir='output_images'):
    # Ensure the output directory exists
    os.makedirs(output_dir, exist_ok=True)
    
    # Calculate the total number of images to display
    total_images = images_per_row * rows
    
    # Filter for the specified range of images
    filtered_df = df.dropna(subset=['image_path', 'x', 'y'])
    image_paths = filtered_df['image_path'].iloc[:total_images].tolist()
    x_coords = filtered_df['x'].iloc[:total_images].tolist()
    y_coords = filtered_df['y'].iloc[:total_images].tolist()

    # Initialize the figure
    fig, axes = plt.subplots(rows, images_per_row, figsize=(20, rows * 4))
    axes = axes.flatten() if rows * images_per_row > 1 else [axes]

    for i, (image_path, x, y) in enumerate(zip(image_paths, x_coords, y_coords)):
        # Load the DICOM image using pydicom
        dicom_image = pydicom.dcmread(image_path)
        image_array = dicom_image.pixel_array

        # Display the image
        axes[i].imshow(image_array, cmap="gray")
        axes[i].scatter(x, y, color='red', s=30)  # Overlay point with red marker
        axes[i].axis('off')

        # Add severity and level as the title
        severity = filtered_df.iloc[i]['severity']
        level = filtered_df.iloc[i]['level']
        axes[i].set_title(f"Image: {image_path}\nPoint: ({x:.2f}, {y:.2f})\nSeverity: {severity}\nLevel: {level}", fontsize=8)
        
        # Save the image with annotations as a JPG file
        fig_save, ax_save = plt.subplots()
        ax_save.imshow(image_array, cmap="gray")
        ax_save.scatter(x, y, color='red', s=30)
        ax_save.axis('off')
        ax_save.set_title(f"Point: ({x:.2f}, {y:.2f})\nSeverity: {severity}\nLevel: {level}", fontsize=8)
        
        # Save as JPG
        output_path = os.path.join(output_dir, f"image_{i + 1}.jpg")
        fig_save.savefig(output_path, bbox_inches='tight', dpi=300)
        plt.close(fig_save)  # Close the figure to free up memory

    plt.tight_layout()
    plt.show()

# Example usage:
# Assuming end_full_df contains 'image_path', 'x', 'y', 'severity', and 'level' columns.
display_images_with_annotations(end_full_df, images_per_row=1, rows=25)

Explanation of Sagittal T2 MRI Sequence:

The Sagittal T2 sequence is a key imaging modality in spine MRI because it provides clear, detailed views of the spinal canal and surrounding structures. Hereâs why it is particularly useful:

1. T2-Weighted Imaging Highlights Fluid:

  • T2-weighted MRI images are specifically designed to make fluids appear bright. In the spine, this is crucial because:
    • The cerebrospinal fluid (CSF) surrounding the spinal cord appears as a bright signal.
    • Structures like the spinal cord and nerve roots stand out against the bright CSF background, enabling better contrast.

2. Visualization of the Spinal Canal:

  • The spinal canal is the space through which the spinal cord and nerve roots pass. Using a Sagittal T2 sequence:
    • Any narrowing (stenosis) of the canal can be easily detected because the contrast between the CSF (bright) and compressed areas (dark) is very apparent.
    • Herniated discs or other abnormalities that impinge on the canal or nerves become clearly visible.

3. Sagittal Plane Provides a Longitudinal View:

  • Sagittal views are taken from the side, offering a complete longitudinal view of the entire spine, including:
    • Vertebrae
    • Intervertebral discs
    • Ligaments
    • Spinal cord and CSF
  • This broad perspective allows for the identification of structural issues, such as disc degeneration or misalignments, that might affect the spinal canal.

4. Assessment of Narrowing (Stenosis):

  • Conditions like spinal canal stenosis or neural foraminal narrowing can be effectively assessed. With T2-weighting:
    • Compressed areas or regions with reduced CSF appear dark, surrounded by the bright CSF.
    • This highlights any areas where the spinal cord or nerve roots might be at risk.

5. Clinical Applications:

  • Disc Herniation: Disc material pressing against the spinal cord or nerve roots can be seen.
  • Spinal Stenosis: Narrowing of the spinal canal is well visualized.
  • Tumors or Cysts: Masses that displace the CSF or compress the cord are easily detectable.
  • Inflammatory Conditions: Swelling or edema in the spine structures shows as bright or dark regions depending on fluid content.

Summary:

In a Sagittal T2-weighted image, the ability to contrast the bright CSF against the darker spinal cord, vertebrae, and discs makes it an invaluable tool for assessing spinal canal health and identifying narrowing or compression that might lead to neurological symptoms. This sequence is typically the first choice for evaluating degenerative spine conditions or identifying potential nerve impingements.

import os

def display_images_with_annotations(df, images_per_row=1, rows=1):
    # Calculate the total number of images to display
    total_images = images_per_row * rows
    
    # Filter for valid DICOM images and exclude non-DICOM files
    filtered_df = df.dropna(subset=['image_path', 'x', 'y'])
    filtered_df = filtered_df[~filtered_df['image_path'].str.endswith('.jpg')]  # Exclude .jpg files

    image_paths = filtered_df['image_path'].iloc[:total_images].tolist()
    x_coords = filtered_df['x'].iloc[:total_images].tolist()

    # Initialize the figure
    fig, axes = plt.subplots(rows, images_per_row, figsize=(20, rows * 4))

        except Exception as e:
            print(f"Skipping image {image_path} due to error: {e}")
            continue

    plt.tight_layout()
    plt.show()
